# Запускаем контейнер с базой данных MySQL
docker run \
       --name DB-mysql \
       -e MYSQL_ROOT_PASSWORD=pw1234 \
       -d mysql:latest

# Подключаемся к этому контейнеру:
# флаг "-t" выделяет pts (виртуальный телетайп), или, другими словами, сессию терминала, который подключен к потоку ввода контейнера.
# флаг "-i" запускает контейнер в интерактивном режиме.
docker exec -ti DB-mysql /bin/bash

# После того, как зашли в контейнер, смотрим, какие СИСТЕМНЫЕ переменные окружения нам доступны:
env

# Видим, что среди переменных окружения присутствует "MYSQL_ROOT_PASSWORD=pw1234", которую мы задавали при запуске контейнера.

# Запускаем базу данных (указываем пароль pw1234):
mysql -p

# Выходим из БД:
exit

# Также БД можно запустить сразу при выполнении команды exec:
# пароль: pw1234
docker exec -ti DB-mysql mysql -u root -p
